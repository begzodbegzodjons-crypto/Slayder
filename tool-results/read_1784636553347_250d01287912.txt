   484→      id: nextId,

   486→      status: 'Kutmoqda',
   487→      createdAt: new Date().toISOString(),
   488→      isReturning,
   489→      previousVisitId,
   490→      visitCount,
   491→      patientHistory,
   492→    };
   493→
   494→    const updated = [...patients, newPatient];
   495→    savePatientsList(updated);
   496→
   497→    // Dynamic automatic logging of outpatient payment!
   498→    if (newPatient.paymentStatus === 'To\'langan' && newPatient.paymentAmount > 0) {
   499→      const getDeptNameLocal = (id: string) => departments.find((d) => d.id === id)?.name || id;
   500→      const todayDate = new Date().toISOString().split('T')[0];
   501→      const todayTime = new Date().toTimeString().split(' ')[0].substring(0, 5);
   502→
   503→      const newTx: ClinicTransaction = {
   504→        id: 'TX-' + Math.floor(Math.random() * 90000 + 10000),
   505→        type: 'Kirim',
   506→        amount: newPatient.paymentAmount,
   507→        category: "Ambulator ko'rik",
   508→        description: `${newPatient.lastName} ${newPatient.firstName} - ${getDeptNameLocal(newPatient.departmentId)} bo'limiga navbat to'lovi`,
   509→        date: todayDate,
   510→        time: todayTime,
   511→        createdAt: new Date().toISOString(),
   512→        patientId: newPatient.id,
   513→        patientName: `${newPatient.lastName} ${newPatient.firstName}`
   514→      };
   515→
   516→      const updatedTx = [newTx, ...transactions];
   517→      saveTransactionsList(updatedTx);
   518→    }
   519→  };
   520→
   521→  // Toggle patient's payment status
   522→  const handleUpdatePaymentStatus = (patientId: string, status: 'To\'langan' | 'Kutilmoqda') => {
   523→    const updated = patients.map((p) => {
   524→      if (p.id === patientId) {
   525→        return { ...p, paymentStatus: status };
   526→      }
   527→      return p;
   528→    });
   529→    savePatientsList(updated);
   530→
   531→    // If changing to 'To'langan', log a transaction!
   532→    const patient = patients.find(p => p.id === patientId);
   533→    if (patient && status === 'To\'langan' && patient.paymentAmount > 0) {
   534→      const isAlreadyLogged = transactions.some(tx => tx.patientId === patientId && tx.category === "Ambulator ko'rik");
   535→      if (!isAlreadyLogged) {
   536→        const getDeptNameLocal = (id: string) => departments.find((d) => d.id === id)?.name || id;
   537→        const todayDate = new Date().toISOString().split('T')[0];
   538→        const todayTime = new Date().toTimeString().split(' ')[0].substring(0, 5);
   539→
   540→        const newTx: ClinicTransaction = {
   541→          id: 'TX-' + Math.floor(Math.random() * 90000 + 10000),
   542→          type: 'Kirim',
   543→          amount: patient.paymentAmount,
   544→          category: "Ambulator ko'rik",
   545→          description: `${patient.lastName} ${patient.firstName} - ${getDeptNameLocal(patient.departmentId)} bo'limiga navbat to'lovi`,
   546→          date: todayDate,
   547→          time: todayTime,
   548→          createdAt: new Date().toISOString(),
   549→          patientId: patient.id,
   550→          patientName: `${patient.lastName} ${patient.firstName}`
   551→        };
   552→        saveTransactionsList([newTx, ...transactions]);
   553→      }
   554→    }
   555→  };
   556→
   557→  // Call / Accept Patient to Doctor Cabinet (turns status to 'Qabulda' and alerts TV Monitor)
   558→  const handleCallPatient = (calledPatient: Patient) => {

   560→    const updated = patients.map((p) => {

   562→        return {
   563→          ...p,
   564→          status: 'Qabulda' as const,
   565→          calledAt: new Date().toISOString(),

   567→      }

   569→      if (p.departmentId === calledPatient.departmentId && p.status === 'Qabulda') {
   570→        return {
   571→          ...p,
   572→          status: 'Yakunlangan' as const,
   573→          completedAt: new Date().toISOString(),
   574→          diagnosis: p.diagnosis || 'Ko\'rik yakunlandi',
   575→        };
   576→      }
   577→      return p;
   578→    });

   580→    savePatientsList(updated);

   587→        return { ...p, ...updates };

   590→    });

   605→  // To'lov qaytarish (refund) - bemor to'lov qilgan, lekin davolanishdan bosh tortgan

   607→  // 1. Bemor holatini "Bekor qilingan" ga o'zgartiradi
   608→  // 2. refundStatus = 'Qaytarildi' qiladi
   609→  // 3. Kassaga Chiqim (xarajat) tranzaksiyasini qo'shadi - "To'lov qaytarildi" kategoriyasi
   610→  // 4. Barcha hisobotlarda avtomatik aks etadi
   611→  const handleRefundPatient = (
   612→    patientId: string,
   613→    refundAmount: number,
   614→    reason: string
   615→  ) => {
   616→    const patient = patients.find((p) => p.id === patientId);
   617→    if (!patient) {
   618→      alert('Bemor topilmadi!');
   619→      return;
   620→    }
   621→
   622→    if (patient.paymentStatus !== 'To\'langan') {
   623→      alert('Ushbu bemor to\'lov qilmagan! To\'lov qaytarib bo\'lmaydi.');
   624→      return;
   625→    }
   626→
   627→    if (refundAmount <= 0 || refundAmount > patient.paymentAmount) {
   628→      alert(
   629→        `Noto'g'ri summa! To'lov qaytarish summasi 0 dan katta va ${patient.paymentAmount.toLocaleString()} UZS dan oshmasligi kerak.`
   630→      );
   631→      return;
   632→    }
   633→
   634→    // 1. Bemor ma'lumotlarini yangilash
   635→    const updatedPatients = patients.map((p) => {
   636→      if (p.id === patientId) {
   637→        return {
   638→          ...p,
   639→          status: 'Bekor qilingan' as const,
   640→          refundStatus: (refundAmount === p.paymentAmount ? 'Qaytarildi' : 'Qisman') as 'Qaytarildi' | 'Qisman',
   641→          refundedAmount: refundAmount,
   642→          refundedAt: new Date().toISOString(),
   643→          refundedReason: reason.trim(),
   644→        };
   645→      }

   647→    });
   648→    savePatientsList(updatedPatients);
   649→
   650→    // 2. Kassaga Chiqim (xarajat) tranzaksiyasini qo'shish
   651→    const todayDate = new Date().toISOString().split('T')[0];
   652→    const todayTime = new Date().toTimeString().split(' ')[0].substring(0, 5);
   653→    const getDeptNameLocal = (id: string) => departments.find((d) => d.id === id)?.name || id;
   654→
   655→    const refundTx: ClinicTransaction = {
   656→      id: 'TX-' + Math.floor(Math.random() * 90000 + 10000),
   657→      type: 'Chiqim',
   658→      amount: refundAmount,
   659→      category: "To'lov qaytarildi",
   660→      description: `${patient.lastName} ${patient.firstName} - ${getDeptNameLocal(patient.departmentId)} bo'limi uchun to'lov qaytarildi. Sabab: ${reason.trim() || 'Ko\'rsatilmagan'}`,
   661→      date: todayDate,
   662→      time: todayTime,
   663→      createdAt: new Date().toISOString(),
   664→      patientId: patient.id,
   665→      patientName: `${patient.lastName} ${patient.firstName}`,
   666→    };
   667→
   668→    const updatedTx = [refundTx, ...transactions];
   669→    saveTransactionsList(updatedTx);
   670→  };
   671→
   672→  // Open the HDMI TV queue monitor in a separate tab
   673→  const handleOpenMonitorWindow = () => {

   677→      alert('Tashqi oyna ochish brauzer tomonidan bloklandi! Iltimos, manzil satridagi block belgisi orqali popup ruxsatini bering.');
   678→    }
   679→  };
   680→
   681→  // If this tab is opened specifically for the external TV (via HDMI)

   684→  }
   685→

   688→    return (
   689→      <LoginPage
   690→        departments={departments}
   691→        receptionStaff={receptionStaff}
   692→        onLoginSuccess={handleLoginSuccess}
   693→        openMonitorWindow={handleOpenMonitorWindow}

   696→  }
   697→
   698→  return (

   719→            setPatients={savePatientsList}
   720→            hospitalRooms={hospitalRooms}

   726→            setReceptionStaff={saveReceptionStaffList}

   768→              <div className="flex items-start space-x-4">
